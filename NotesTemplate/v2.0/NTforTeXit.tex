\newcommand{\abisect}[6][] % Source: https://tex.stackexchange.com/questions/233246/create-angle-bisectors-in-tikz
{
	\path[#1] let
	\p1 = ($(#3)!1cm!(#2)$),
	\p2 = ($(#3)!1cm!(#4)$),
	\p3 = ($(\p1) + (\p2) - (#3)$)
	in
	($(#3)!#6!(\p3)$) -- ($(\p3)!#5!(#3)$) ;
}
